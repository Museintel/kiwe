// ==================== UPLOAD HANDLER ====================
function seam_ingest_handle_upload(): void
{
    if (!current_user_can('manage_options')) {
        seam_ingest_notice('error', 'Permission denied.');
        return;
    }
    if (empty($_FILES['seam_project_file']) || $_FILES['seam_project_file']['error'] !== UPLOAD_ERR_OK) {
        seam_ingest_notice('error', 'Please upload a valid seam.project.json file.');
        return;
    }

    $json = file_get_contents($_FILES['seam_project_file']['tmp_name']);
    $project = json_decode($json, true);

    if (!is_array($project) || ($project['seam'] ?? '') !== '3.0') {
        seam_ingest_notice('error', 'Invalid Seam v3 project file. Must contain "seam": "3.0".');
        return;
    }

    wp_mkdir_p(SEAM_UPLOAD_DIR);
    file_put_contents(SEAM_PROJECT_FILE, $json);

    $result = seam_ingest_project($project, md5($json), !empty($_POST['seam_force']), $json);
    seam_ingest_notice($result['success'] ? 'success' : 'error', $result['message']);
}

function seam_ingest_handle_admin_actions(): void
{
    if (!current_user_can('manage_options')) return;

    if (isset($_POST['seam_reingest_saved'], $_POST['seam_reingest_nonce']) && wp_verify_nonce($_POST['seam_reingest_nonce'], 'seam_reingest_saved')) {
        $file = SEAM_UPLOAD_DIR . '/seam.project.json';
        if (file_exists($file)) {
            $json = file_get_contents($file);
            $project = json_decode($json, true);
            if (is_array($project)) seam_ingest_project($project, md5($json), true, $json);
        }
    }

    if (isset($_POST['seam_clear_ingest'], $_POST['seam_clear_nonce']) && wp_verify_nonce($_POST['seam_clear_nonce'], 'seam_clear_ingest')) {
        // clear logic (kept minimal for snippet)
        delete_option('seam_project_state');
        delete_option('seam_ingest_hash');
    }
}

// ==================== CORE INGEST LOGIC (UPDATED FOR v3) ====================
function seam_ingest_project(array $project, string $hash, bool $force = false, string $raw_json = ''): array
{
    if (!wp_mkdir_p(SEAM_UPLOAD_DIR)) {
        return ['success' => false, 'message' => 'Could not create upload folder.'];
    }

    // Generate full computed tokens from minimal project.json
    $computed = seam_generate_full_computed_tokens($project);

    // Build Bricks data
    $variables = seam_build_bricks_variables_from_computed($computed, $project);
    $palette = seam_build_bricks_palette($project);

    // Write generated token files
    file_put_contents(SEAM_TOKENS_JSON, wp_json_encode($computed, JSON_PRETTY_PRINT | JSON_UNESCAPED_SLASHES));
    file_put_contents(SEAM_TOKENS_GENERATED_CSS, seam_generate_tokens_generated_css($computed, $project));

    // Write runtime data
    $runtime_data = [
        'project' => $project,
        'personality' => $project['personality'] ?? [],
        'brand' => $project['brand'] ?? []
    ];
    file_put_contents(SEAM_RUNTIME_DATA, wp_json_encode($runtime_data, JSON_PRETTY_PRINT | JSON_UNESCAPED_SLASHES));

    // Update Bricks options
    update_option(BRICKS_DB_GLOBAL_VARIABLES, $variables);
    update_option(BRICKS_DB_COLOR_PALETTE, $palette);

    // Save state
    update_option('seam_project_state', [
        'project' => $project,
        'ingested_at' => current_time('mysql')
    ]);
    update_option('seam_ingest_hash', $hash);

    return ['success' => true, 'message' => 'Seam v3 project ingested successfully. Tokens and Bricks variables generated.'];
}

function seam_generate_tokens_generated_css(array $computed, array $project): string
{
    $css = "/* SEAM TOKENS GENERATED CSS — v3.0 (auto-generated by ingest.php) */\n";
    $css .= "@layer seam.tokens {\n:root {\n";

    foreach ($computed as $group => $tokens) {
        if (!is_array($tokens)) continue;
        foreach ($tokens as $name => $value) {
            if (is_array($value)) continue; // skip heading_hierarchy etc.
            $css .= "    {$name}: {$value};\n";
        }
    }

    // Dark mode (basic)
    $css .= "}\n\n:root[data-theme=\"dark\"] {\n";
    $css .= "    --color-surface: #151816;\n";
    $css .= "    --color-surface-raised: #1E241F;\n";
    $css .= "    --color-text: #F2F4F1;\n";
    $css .= "}\n}\n";

    return $css;
}

function seam_build_bricks_variables_from_computed(array $computed, array $project): array
{
    $vars = [];
    foreach ($computed as $group => $tokens) {
        if (!is_array($tokens)) continue;
        foreach ($tokens as $name => $value) {
            if (is_array($value)) continue;
            $vars[] = [
                'name' => $name,
                'value' => $value,
                'category' => 'Seam' . ucfirst($group)
            ];
        }
    }
    return $vars;
}

function seam_build_bricks_palette(array $project): array
{
    $colors = $project['brand']['colors'] ?? [];
    $palette = [];
    foreach ($colors as $name => $hex) {
        $palette[] = ['hex' => $hex];
    }
    return [['id' => 'seam', 'name' => 'Seam', 'colors' => $palette]];
}

function seam_ingest_notice(string $type, string $message): void
{

    echo '<div class="notice ' . esc_attr($class) . ' is-dismissible"><p>' . wp_kses_post($message) . '</p></div>';
}
